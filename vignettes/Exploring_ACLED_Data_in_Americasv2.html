events <- levels(as.factor(acled.data$event_type))
for (l in events) {
p <-  acled.data  %>%
  filter(., event_type == l)  %>%
  filter(., region == "South America" )  %>%

 }

Frequency Plot for Number of Conflicts

for (l in events) {
  cat(l)
  p <-  acled.data %>%
    filter(., event_type == l)  %>%
    filter(., region == "South America" )  %>%
    filter(., ! ( country %in% c("French Guiana", "Falkland Islands")))  %>%
    group_by(yearmonth, country)%>%
    summarise(count=n(),
              fatalities =  sum(fatalities))%>%
    arrange(desc(count)) %>%
    ggplot(aes(x = yearmonth ,
               y=count)) +
    geom_bar(stat = "identity",
             color= unhcr_blue,
             fill = unhcr_blue) +
    scale_y_continuous(name = "Total Number of Conflicts",
                       labels = scales::comma) +
    labs(title = paste0("Frequency Plot for Number of Conflicts in relation with ",l),
            subtitle = "Data about number of conflicts which happened",
            caption = "Acled Data") +
    facet_wrap(~ country, ncol = 3) +
    unhcr_theme()
  print(p)
}
